46	pub fn inputs(&self) -> Result<Vec<Outlet>> {
47		let mut num_inputs = 0;
48		ortsys![unsafe KernelInfo_GetInputCount(self.ptr.as_ptr(), &mut num_inputs)?];
49
50		let mut inputs = Vec::with_capacity(num_inputs);
51		for idx in 0..num_inputs {
52			let mut name_len = 0;
53			ortsys![unsafe KernelInfo_GetInputName(self.ptr.as_ptr(), idx, ptr::null_mut(), &mut name_len)?];
54			let mut name = vec![0u8; name_len];
55			ortsys![unsafe KernelInfo_GetInputName(self.ptr.as_ptr(), idx, name.as_mut_ptr().cast::<c_char>(), &mut name_len)?];
56			let name = CString::from_vec_with_nul(name)?.into_string()?;
57			let mut type_info = ptr::null_mut();
58			ortsys![unsafe KernelInfo_GetInputTypeInfo(self.ptr.as_ptr(), idx, &mut type_info)?; nonNull(type_info)];
59			let dtype = unsafe { ValueType::from_type_info(type_info) };
60			inputs.push(Outlet::new(name, dtype))
61		}
62		Ok(inputs)
63	}

84	pub fn constant_input<T: DowncastableTarget>(&self, idx: usize) -> Result<ValueRef<'_, T>> {
85		let mut value_ptr: *const ort_sys::OrtValue = ptr::null();
86		let mut is_constant = 0;
87		ortsys![unsafe KernelInfoGetConstantInput_tensor(self.ptr.as_ptr(), idx, &mut is_constant, &mut value_ptr)?];
88		if is_constant == 0 {
89			return Err(Error::new("input is not constant"));
90		}
91
92		let Some(value_ptr) = NonNull::new(value_ptr.cast_mut()) else {
93			return Err(Error::new("input index out of bounds"));
94		};
95
96		unsafe { ValueRef::new(DynValue::from_ptr_nodrop(value_ptr, None)) }.downcast()
97	}

148pub struct ScratchBuffer<T> {
149	allocator: Allocator,
150	buffer: *mut T,
151	size: usize
152}
153
154impl<T> ScratchBuffer<T> {
155	pub fn as_slice(&self) -> Option<&[T]> {
156		if self.allocator.memory_info().is_cpu_accessible() {
157			Some(unsafe { slice::from_raw_parts(self.buffer.cast_const(), self.size) })
158		} else {
159			None
160		}
161	}
162
163	pub fn as_mut_slice(&mut self) -> Option<&mut [T]> {
164		if self.allocator.memory_info().is_cpu_accessible() {
165			Some(unsafe { slice::from_raw_parts_mut(self.buffer, self.size) })
166		} else {
167			None
168		}
169	}
170}
171
172impl<T> Drop for ScratchBuffer<T> {
173	fn drop(&mut self) {
174		unsafe {
175			self.allocator.free(self.buffer);
176		}
177		crate::logging::drop!(ScratchBuffer, self.buffer);
178	}
179}
