1//! The [`Environment`] is a process-global configuration under which [`Session`](crate::session::Session)s are created.
2//!
3//! With it, you can configure [default execution providers], enable/disable [telemetry], share a [global thread pool]
4//! across all sessions, or add a [custom logger].
5//!
6//! Environments can be set up via [`ort::init`](init):
7//! ```
8//! # use ort::ep;
9//! # fn main() -> ort::Result<()> {
10//! ort::init().with_execution_providers([ep::CUDA::default().build()]).commit();
11//!
12//! // ... do other ort things now that our environment is set up...
13//! # Ok(())
14//! # }
15//! ```
16//!
17//! With the `load-dynamic` feature, you can also load the runtime from a direct path to a DLL with
18//! [`ort::init_from`](init_from):
19//!
20//! ```ignore
21//! # use ort::ep;
22//! # fn main() -> ort::Result<()> {
23//! let lib_path = std::env::current_exe().unwrap().parent().unwrap().join("lib");
24//! ort::init_from(lib_path.join("onnxruntime.dll"))?
25//! 	.with_execution_providers([ep::CUDA::default().build()])
26//! 	.commit();
27//! # Ok(())
28//! # }
29//! ```
30//!
31//! If you don't configure an environment, one will be created with default settings at the first creation of a session.
32//! The environment can't be re-configured after one is committed, so it's important `ort::init` come before any other
33//! `ort` API for the config to take effect. Authors of libraries using `ort` should **never** have the library
34//! configure the environment itself; allow the application developer to do that themselves if they wish.
35//!
36//! [default execution providers]: EnvironmentBuilder::with_execution_providers
37//! [telemetry]: EnvironmentBuilder::with_telemetry
38//! [global thread pool]: EnvironmentBuilder::with_global_thread_pool
39//! [custom logger]: EnvironmentBuilder::with_logger

67// Rust doesn't run destructors for statics, but ONNX Runtime is *very* particular about `ReleaseEnv` being called
68// before any C++ destructors are called. In order to drop the environment, we have to release the reference held in
69// `G_ENV` at the end of the program, but before C++ destructors are called. On Linux & Windows (surprisingly), this is
70// fairly simple: just put it in a custom linker section.
71//
72// `G_ENV` used to be `Mutex<Weak<Environment>>`, which was much nicer, but apparently you can only ever call
73// `CreateEnv` once throughout the lifetime of the process, *even if* the last env was `ReleaseEnv`'d. So once all
74// `Session`s fell out of scope, if you ever tried to create another one, you'd crash. Grand.
75#[cfg_attr(any(target_os = "linux", target_os = "android"), unsafe(link_section = ".text.exit"))]
76unsafe extern "C" fn release_env_on_exit(#[cfg(target_vendor = "apple")] _: *const ()) {
77	G_ENV.lock().take();
78}
79
80#[used]
81#[cfg(all(not(windows), not(target_vendor = "apple"), not(target_arch = "wasm32")))]
82#[unsafe(link_section = ".fini_array")]
83static _ON_EXIT: unsafe extern "C" fn() = release_env_on_exit;
84#[used]
85#[cfg(windows)]
86#[unsafe(link_section = ".CRT$XLB")]
87static _ON_EXIT: unsafe extern "system" fn(module: *mut (), reason: u32, reserved: *mut ()) = {
88	unsafe extern "system" fn on_exit(_h: *mut (), reason: u32, _pv: *mut ()) {
89		// XLB gets called on both init & exit (?). Also, XLA never gets called, and no one online ever mentions that.
90		// Only do destructor things if we're actually exiting the process (DLL_PROCESS_EXIT = 0)
91		if reason == 0 {
92			unsafe { release_env_on_exit() };
93		}
94	}
95	on_exit
96};

98// macOS used to have the __mod_term_func section which worked similar to `.fini_array`, but one day they just decided
99// to remove it I guess? So we have to set an atexit handler instead. But normal atexit doesn't work, we need to use
100// __cxa_atexit. And if you register it too early in the program (i.e. in __mod_init_func), it'll fire *after* C++
101// destructors. So we call this after we create the environment instead. This shit took years off my life.
102#[cfg(target_vendor = "apple")]
103fn register_atexit() {
104	unsafe extern "C" {
105		static __dso_handle: *const ();
106		fn __cxa_atexit(cb: unsafe extern "C" fn(_: *const ()), arg: *const (), dso_handle: *const ());
107	}
108	unsafe { __cxa_atexit(release_env_on_exit, core::ptr::null(), __dso_handle) };
109}

159	/// Registers an execution provider library from the given `path`. Can be used to customize the path of a provider
160	/// library, or load new ones ONNX Runtime was not initially compiled with.
161	///
162	/// `name` is semi-arbitrary - it should be unique per EP library. Adding the suffix `.virtual` to `name` allows the
163	/// EP library to create virtual [devices](crate::device).
164	///
165	/// Returns a handle that can be used to [unregister](ExecutionProviderLibrary::unregister) the library, should it
166	/// no longer be needed.
167	///
168	/// ```
169	/// # use ort::environment::Environment;
170	/// # fn main() -> ort::Result<()> {
171	/// let env = Environment::current()?;
172	///
173	/// let _ = env.register_ep_library("CUDA", "/path/to/onnxruntime_providers_cuda.dll");
174	/// # Ok(())
175	/// # }
176	/// ```

179	pub fn register_ep_library<P: AsRef<Path>>(self: &Arc<Self>, name: impl Into<String>, path: P) -> Result<ExecutionProviderLibrary> {
180		let name = name.into();
181		let path = crate::util::path_to_os_char(path);
182		with_cstr(name.as_bytes(), &|name| {
183			ortsys![unsafe RegisterExecutionProviderLibrary(self.ptr().cast_mut(), name.as_ptr(), path.as_ptr())?];
184			Ok(())
185		})?;
186		Ok(ExecutionProviderLibrary::new(name, self))
187	}

189	/// Returns an iterator over all automatically discovered [hardware device](crate::device::Device)s.
190	///
191	/// ```
192	/// # use ort::environment::Environment;
193	/// # fn main() -> ort::Result<()> {
194	/// let env = Environment::current()?;
195	/// for device in env.devices() {
196	/// 	println!(
197	/// 		"{id} ({vendor} {ty:?} - {ep})",
198	/// 		id = device.id(),
199	/// 		vendor = device.vendor()?,
200	/// 		ty = device.ty(),
201	/// 		ep = device.ep()?
202	/// 	);
203	/// }
204	/// # Ok(())
205	/// # }
206	/// ```
207	#[cfg(feature = "api-22")]
208	#[cfg_attr(docsrs, doc(cfg(feature = "api-22")))]
209	pub fn devices(&self) -> impl DoubleEndedIterator<Item = crate::device::Device<'_>> + '_ {
210		let mut ptrs = ptr::dangling();
211		let mut len = 0;
212		// returns an error in minimal build because its unsupported. ignore & return empty iterator in that case
213		let _ = ortsys![@ort: unsafe GetEpDevices(self.ptr().cast_mut(), &mut ptrs, &mut len) as Result];
214		unsafe { core::slice::from_raw_parts(ptrs, len) }
215			.iter()
216			.filter_map(|c| NonNull::new(c.cast_mut()))
217			.map(crate::device::Device::new)
218	}

247/// Returns a handle to the currently active `Environment`. If one has not yet been committed (or an old environment
248/// has fallen out of usage), a new environment will be created & committed.
249pub fn current() -> Result<Arc<Environment>> {
250	let mut env_lock = G_ENV.lock();
251	if let Some(env) = env_lock.as_ref() {
252		return Ok(env.clone());
253	}
254
255	let options = G_ENV_OPTIONS.get_or_init(EnvironmentBuilder::new);
256	let env = options.create_environment().map(Arc::new)?;
257	*env_lock = Some(Arc::clone(&env));
258
259	#[cfg(target_vendor = "apple")]
260	register_atexit();
261
262	Ok(env)
263}

309		ortsys![unsafe SetGlobalSpinControl(self.ptr_mut(), if spin_control { 1 } else { 0 })?];
310		Ok(self)
311	}

314		let ptr = self.ptr_mut();
315		with_cstr(affinity.as_ref().as_bytes(), &|affinity| {
316			ortsys![unsafe SetGlobalIntraOpThreadAffinity(ptr, affinity.as_ptr())?];
317			Ok(())
318		})?;
319		Ok(self)
320	}

322	/// Disables subnormal floats by enabling the denormals-are-zero and flush-to-zero flags for all threads in the
323	/// pool.
324	///
325	/// [Subnormal floats](https://en.wikipedia.org/wiki/Subnormal_number) are extremely small numbers very close to zero.
326	/// Operations involving subnormal numbers can be very slow; enabling this flag will instead treat them as `0.0`,
327	/// giving faster & more consistent performance, but lower accuracy (in cases where subnormals are involved).
328	pub fn with_flush_to_zero(mut self) -> Result<Self> {
329		ortsys![unsafe SetGlobalDenormalAsZero(self.ptr_mut())?];
330		Ok(self)
331	}

333	/// Use a custom [thread manager](ThreadManager) to spawn threads for the global thread pool.
334	pub fn with_thread_manager<T: ThreadManager + Any + 'static>(mut self, manager: T) -> Result<Self> {
335		let manager = Arc::new(manager);
336		ortsys![unsafe SetGlobalCustomThreadCreationOptions(self.ptr_mut(), (&*manager as *const T as *mut T).cast())?];
337		ortsys![unsafe SetGlobalCustomCreateThreadFn(self.ptr_mut(), Some(thread_create::<T>))?];
338		ortsys![unsafe SetGlobalCustomJoinThreadFn(self.ptr_mut(), Some(thread_join::<T>))?];
339		self.thread_manager = Some(manager as Arc<dyn Any>);
340		Ok(self)
341	}

359/// Used for customizing the thread spawning process of a [global thread pool](GlobalThreadPoolOptions) or [session
360/// thread pool][session]. Could be used to add additional initialization/cleanup code to inference threads for
361/// better debugging/error handling.
362///
363/// Threads spawned by `ThreadManager` should be *real* threads, spawned directly via the operating system; they
364/// shouldn't be spawned in another thread pool like [`rayon`](https://crates.io/crates/rayon) because sessions have
365/// their own (interfering) thread pool logic.
366///
367/// A very simple thread manager would be:
368/// ```
369/// use std::thread::{self, JoinHandle};
370///
371/// use ort::environment::ThreadManager;
372///
373/// struct StdThreadManager;
374///
375/// impl ThreadManager for StdThreadManager {
376/// 	type Thread = JoinHandle<()>;
377///
378/// 	fn create(&self, work: impl FnOnce() + Send + 'static) -> ort::Result<Self::Thread> {
379/// 		Ok(thread::spawn(move || {
380/// 			// ... maybe optional initialization code ...
381///
382/// 			// threads must call work() to actually do the work the runtime needs
383/// 			work();
384///
385/// 			// ... maybe optional destructor code ...
386/// 		}))
387/// 	}
388///
389/// 	fn join(thread: Self::Thread) -> ort::Result<()> {
390/// 		let _ = thread.join();
391/// 		Ok(())
392/// 	}
393/// }
394/// ```
395///
396/// [session]: crate::session::builder::SessionBuilder::with_thread_manager
397pub trait ThreadManager {
398	/// A handle to a spawned thread; used to [`join`](ThreadManager::join) it later.
399	type Thread;
400
401	/// Spawns a thread.
402	///
403	/// The newly spawned thread must call `work()`.
404	fn create(&self, work: impl FnOnce() + Send + 'static) -> crate::Result<Self::Thread>;
405
406	/// Wait for the thread to finish, like [`std::thread::JoinHandle::join`].
407	fn join(thread: Self::Thread) -> crate::Result<()>;
408}
409
410pub(crate) unsafe extern "system" fn thread_create<T: ThreadManager + Any>(
411	ort_custom_thread_creation_options: *mut c_void,
412	ort_thread_worker_fn: ort_sys::OrtThreadWorkerFn,
413	ort_worker_fn_param: *mut c_void
414) -> ort_sys::OrtCustomThreadHandle {
415	struct SendablePtr(*mut c_void);
416	unsafe impl Send for SendablePtr {}
417
418	let ort_worker_fn_param = SendablePtr(ort_worker_fn_param);
419
420	let runner = || {
421		let manager = unsafe { &mut *ort_custom_thread_creation_options.cast::<T>() };
422		<T as ThreadManager>::create(manager, move || {
423			let p = ort_worker_fn_param;
424			unsafe { (ort_thread_worker_fn)(p.0) }
425		})
426	};
427	#[cfg(not(feature = "std"))]
428	let res = Result::<_, crate::Error>::Ok(runner()); // dumb hack
429	#[cfg(feature = "std")]
430	let res = std::panic::catch_unwind(runner);
431	match res {
432		Ok(Ok(thread)) => (Box::leak(Box::new(thread)) as *mut <T as ThreadManager>::Thread)
433			.cast_const()
434			.cast::<ort_sys::OrtCustomHandleType>(),
435		Ok(Err(e)) => {
436			crate::error!("Failed to create thread using manager: {e}");
437			let _ = e;
438			ptr::null()
439		}
440		Err(e) => {
441			crate::error!("Thread manager panicked: {e:?}");
442			let _ = e;
443			ptr::null()
444		}
445	}
446}

448pub(crate) unsafe extern "system" fn thread_join<T: ThreadManager + Any>(ort_custom_thread_handle: ort_sys::OrtCustomThreadHandle) {
449	let handle = unsafe { Box::from_raw(ort_custom_thread_handle.cast_mut().cast::<<T as ThreadManager>::Thread>()) };
450	if let Err(e) = <T as ThreadManager>::join(*handle) {
451		crate::error!("Failed to join thread using manager: {e}");
452		let _ = e;
453	}
454}

554	pub(crate) fn create_environment(&self) -> Result<Environment> {
555		let logger = self
556			.logger
557			.as_ref()
558			.map(|c| (crate::logging::custom_logger as ort_sys::OrtLoggingFunction, c as *const _ as *mut c_void));
559		#[cfg(feature = "tracing")]
560		let logger = logger.or(Some((crate::logging::tracing_logger, ptr::null_mut())));
561
562		let env_ptr = with_cstr(self.name.as_bytes(), &|name| {
563			let mut env_ptr: *mut ort_sys::OrtEnv = ptr::null_mut();
564			#[allow(clippy::collapsible_else_if)]
565			if let Some(thread_pool_options) = self.global_thread_pool_options.as_ref() {
566				if let Some((log_fn, log_ptr)) = logger {
567					ortsys![
568						unsafe CreateEnvWithCustomLoggerAndGlobalThreadPools(
569							log_fn,
570							log_ptr,
571							ort_sys::OrtLoggingLevel::ORT_LOGGING_LEVEL_VERBOSE,
572							name.as_ptr(),
573							thread_pool_options.ptr(),
574							&mut env_ptr
575						)?;
576						nonNull(env_ptr)
577					];
578					Ok(env_ptr)
579				} else {
580					ortsys![
581						unsafe CreateEnvWithGlobalThreadPools(
582							crate::logging::default_log_level(),
583							name.as_ptr(),
584							thread_pool_options.ptr(),
585							&mut env_ptr
586						)?;
587						nonNull(env_ptr)
588					];
589					Ok(env_ptr)
590				}
591			} else {
592				if let Some((log_fn, log_ptr)) = logger {
593					ortsys![
594						unsafe CreateEnvWithCustomLogger(
595							log_fn,
596							log_ptr,
597							ort_sys::OrtLoggingLevel::ORT_LOGGING_LEVEL_VERBOSE,
598							name.as_ptr(),
599							&mut env_ptr
600						)?;
601						nonNull(env_ptr)
602					];
603					Ok(env_ptr)
604				} else {
605					ortsys![
606						unsafe CreateEnv(
607							crate::logging::default_log_level(),
608							name.as_ptr(),
609							&mut env_ptr
610						)?;
611						nonNull(env_ptr)
612					];
613					Ok(env_ptr)
614				}
615			}
616		})?;
617
618		let _guard = run_on_drop(|| ortsys![unsafe ReleaseEnv(env_ptr.as_ptr())]);
619
620		if self.telemetry {
621			ortsys![unsafe EnableTelemetryEvents(env_ptr.as_ptr())?];
622		} else {
623			ortsys![unsafe DisableTelemetryEvents(env_ptr.as_ptr())?];
624		}
625
626		forget(_guard);
627
628		crate::logging::create!(Environment, env_ptr);
629		Ok(Environment {
630			execution_providers: self.execution_providers.clone(),
631			ptr: env_ptr,
632			has_global_threadpool: self.global_thread_pool_options.is_some(),
633			_thread_manager: self
634				.global_thread_pool_options
635				.as_ref()
636				.and_then(|options| options.thread_manager.clone()),
637			_logger: self.logger.clone()
638		})
639	}
640
641	/// Commit the environment configuration.
642	///
643	/// Returns `true` if the environment configuration was successfully committed; returns `false` if an environment
644	/// has already been configured, indicating this config will not take effect.
645	pub fn commit(self) -> bool {
646		G_ENV_OPTIONS.try_insert_with(|| self)
647	}
648}

650/// Creates an ONNX Runtime environment.
651///
652/// ```
653/// # use ort::ep;
654/// # fn main() -> ort::Result<()> {
655/// ort::init().with_execution_providers([ep::CUDA::default().build()]).commit();
656/// # Ok(())
657/// # }
658/// ```
659///
660/// # Notes
661/// - It is not required to call this function. If this is not called by the time any other `ort` APIs are used, a
662///   default environment will be created.
663/// - **Library crates that use `ort` shouldn't create their own environment.** Let downstream applications create it.
664/// - In order for environment settings to apply, this must be called **before** you use other APIs like [`Session`],
665///   and you *must* call `.commit()` on the builder returned by this function.
666///
667/// [`Session`]: crate::session::Session
668#[must_use = "commit() must be called in order for the environment to take effect"]
669pub fn init() -> EnvironmentBuilder {
670	EnvironmentBuilder::new()
671}

673/// Creates an ONNX Runtime environment, dynamically loading ONNX Runtime from the library file (`.dll`/`.so`/`.dylib`)
674/// specified by `path`. Returns an error if the dylib fails to load.
675///
676/// This must be called before any other `ort` APIs are used in order for the correct dynamic library to be loaded.
677///
678/// ```no_run
679/// # use ort::ep;
680/// # fn main() -> ort::Result<()> {
681/// let lib_path = std::env::current_exe().unwrap().parent().unwrap().join("lib");
682/// ort::init_from(lib_path.join("onnxruntime.dll"))?
683/// 	.with_execution_providers([ep::CUDA::default().build()])
684/// 	.commit();
685/// # Ok(())
686/// # }
687/// ```
688///
689/// # Notes
690/// - In order for environment settings to apply, this must be called **before** you use other APIs like [`Session`],
691///   and you *must* call `.commit()` on the builder returned by this function.
692///
693/// [`Session`]: crate::session::Session
694#[cfg(all(feature = "load-dynamic", not(target_arch = "wasm32")))]
695#[cfg_attr(docsrs, doc(cfg(feature = "load-dynamic")))]
696#[must_use = "commit() must be called in order for the environment to take effect"]
697pub fn init_from<P: AsRef<std::path::Path>>(path: P) -> Result<EnvironmentBuilder> {
698	crate::load_dylib_from_path(path.as_ref())?;
699	Ok(EnvironmentBuilder::new())
