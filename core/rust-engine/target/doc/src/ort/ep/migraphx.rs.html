6/// [MIGraphX execution provider](https://onnxruntime.ai/docs/execution-providers/MIGraphX-ExecutionProvider.html) for
7/// hardware acceleration with AMD GPUs.
8#[derive(Debug, Clone)]
9pub struct MIGraphX {
10	device_id: i32,
11	enable_fp16: bool,
12	enable_fp8: bool,
13	enable_int8: bool,
14	use_native_calibration_table: bool,
15	int8_calibration_table_name: Option<CString>,
16	save_model_path: Option<CString>,
17	load_model_path: Option<CString>,
18	exhaustive_tune: bool,
19	memory_limit: usize,
20	arena_extend_strategy: ArenaExtendStrategy
21}
22
23impl Default for MIGraphX {
24	fn default() -> Self {
25		Self {
26			device_id: 0,
27			enable_fp16: false,
28			enable_fp8: false,
29			enable_int8: false,
30			use_native_calibration_table: false,
31			int8_calibration_table_name: None,
32			save_model_path: None,
33			load_model_path: None,
34			exhaustive_tune: false,
35			memory_limit: usize::MAX,
36			arena_extend_strategy: ArenaExtendStrategy::NextPowerOfTwo
37		}
38	}
39}

47	/// # use ort::{ep, session::Session};

65	/// # Ok(())

67	/// ```

72	}

75	///

90	/// [`MIGraphX::with_int8_calibration_table`] to be set.

92	/// ```

105	/// Configures the path to the input calibration data for int8 quantization.
106	///
107	/// The `native` parameter specifies the format the calibration data is in - `true` for native int8 format, `false`
108	/// for the JSON dump format.
109	///
110	/// ```
111	/// # use ort::{ep, session::Session};
112	/// # fn main() -> ort::Result<()> {
113	/// let ep = ep::MIGraphX::default().with_int8(true).with_int8_calibration_table("...", false).build();
114	/// # Ok(())
115	/// # }
116	/// ```
117	#[must_use]
118	pub fn with_int8_calibration_table(mut self, table_name: impl AsRef<str>, native: bool) -> Self {
119		self.use_native_calibration_table = native;
120		self.int8_calibration_table_name = Some(CString::new(table_name.as_ref()).expect("invalid string"));
121		self
122	}

185impl ExecutionProvider for MIGraphX {
186	fn name(&self) -> &'static str {
187		"MIGraphXExecutionProvider"
188	}
189
190	fn supported_by_platform(&self) -> bool {
191		cfg!(any(all(target_os = "linux", target_arch = "x86_64"), all(target_os = "windows", target_arch = "x86_64")))
192	}
193
194	#[allow(unused, unreachable_code)]
195	fn register(&self, session_builder: &mut SessionBuilder) -> Result<(), RegisterError> {
196		#[cfg(any(feature = "load-dynamic", feature = "migraphx"))]
197		{
198			use core::ptr;
199
200			use crate::{AsPointer, ortsys};
201
202			let options = ort_sys::OrtMIGraphXProviderOptions {
203				device_id: self.device_id,
204				migraphx_fp16_enable: self.enable_fp16.into(),
205				migraphx_fp8_enable: self.enable_fp8.into(),
206				migraphx_int8_enable: self.enable_int8.into(),
207				migraphx_use_native_calibration_table: self.use_native_calibration_table.into(),
208				migraphx_int8_calibration_table_name: self.int8_calibration_table_name.as_ref().map(|c| c.as_ptr()).unwrap_or_else(ptr::null),
209				migraphx_load_compiled_model: self.load_model_path.is_some().into(),
210				migraphx_load_model_path: self.load_model_path.as_ref().map(|c| c.as_ptr()).unwrap_or_else(ptr::null),
211				migraphx_save_compiled_model: self.save_model_path.is_some().into(),
212				migraphx_save_model_path: self.save_model_path.as_ref().map(|c| c.as_ptr()).unwrap_or_else(ptr::null),
213				migraphx_exhaustive_tune: self.exhaustive_tune,
214				migraphx_mem_limit: self.memory_limit as _,
215				migraphx_arena_extend_strategy: match self.arena_extend_strategy {
216					ArenaExtendStrategy::NextPowerOfTwo => 0,
217					ArenaExtendStrategy::SameAsRequested => 1
218				}
219			};
220			ortsys![unsafe SessionOptionsAppendExecutionProvider_MIGraphX(session_builder.ptr_mut(), &options)?];
221			return Ok(());
222		}
223
224		Err(RegisterError::MissingFeature)
225	}
