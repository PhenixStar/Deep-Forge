76impl<Type: SequenceValueTypeMarker + Sized> Value<Type> {
77	pub fn try_extract_sequence<'s, OtherType: ValueTypeMarker + DowncastableTarget + Debug + Sized>(&'s self) -> Result<Vec<ValueRef<'s, OtherType>>> {
78		match self.dtype() {
79			ValueType::Sequence(_) => {
80				let allocator = Allocator::default();
81
82				let mut len = 0;
83				ortsys![unsafe GetValueCount(self.ptr(), &mut len)?];
84
85				let mut vec = Vec::with_capacity(len);
86				for i in 0..len {
87					let value = extract_from_sequence(self.ptr(), i, &allocator)?;
88
89					let value_type = value.dtype();
90					if !OtherType::can_downcast(value.dtype()) {
91						return Err(Error::new_with_code(
92							ErrorCode::InvalidArgument,
93							format!("Cannot extract Sequence<{}> from {value_type:?}", format_value_type::<OtherType>())
94						));
95					}
96
97					vec.push(value.downcast()?);
98				}
99				Ok(vec)
100			}
101			t => Err(Error::new(format!("Cannot extract Sequence<{}> from {t}", format_value_type::<OtherType>())))
102		}
103	}
104}

106impl<T: ValueTypeMarker + DowncastableTarget + Debug + Sized + 'static> Value<SequenceValueType<T>> {
107	/// Creates a [`Sequence`] from an array of [`Value<T>`].
108	///
109	/// This `Value<T>` must be either a [`Tensor`] or [`Map`].
110	///
111	/// ```
112	/// # use ort::{memory:: Allocator, value::{Sequence, Tensor}};
113	/// # fn main() -> ort::Result<()> {
114	/// # let allocator = Allocator::default();
115	/// let tensor1 = Tensor::<f32>::new(&allocator, [1_usize, 128, 128, 3])?;
116	/// let tensor2 = Tensor::<f32>::new(&allocator, [1_usize, 224, 224, 3])?;
117	/// let value = Sequence::new([tensor1, tensor2])?;
118	///
119	/// for tensor in value.extract_sequence() {
120	/// 	println!("{:?}", tensor.shape());
121	/// }
122	/// # 	Ok(())
123	/// # }
124	/// ```
125	///
126	/// [`Tensor`]: crate::value::Tensor
127	/// [`Map`]: crate::value::Map
128	pub fn new(values: impl IntoIterator<Item = Value<T>>) -> Result<Self> {
129		let mut value_ptr = ptr::null_mut();
130		let values: Vec<Value<T>> = values.into_iter().collect();
131		let value_ptrs: Vec<*const ort_sys::OrtValue> = values.iter().map(|c| c.ptr()).collect();
132		ortsys![
133			unsafe CreateValue(value_ptrs.as_ptr(), values.len(), ort_sys::ONNXType::ONNX_TYPE_SEQUENCE, &mut value_ptr)?;
134			nonNull(value_ptr)
135		];
136		Ok(Value {
137			inner: ValueInner::new_backed(
138				value_ptr,
139				// 1. `CreateValue` enforces that we have at least 1 value
140				// 2. `CreateValue` internally uses the first value to determine the element type, so we do the same here
141				ValueType::Sequence(Box::new(values[0].inner.dtype.clone())),
142				None,
143				true,
144				Box::new(values)
145			),
146			_markers: PhantomData
147		})
148	}
149}

151impl<T: ValueTypeMarker + DowncastableTarget + Debug + Sized> Value<SequenceValueType<T>> {
152	pub fn extract_sequence<'s>(&'s self) -> Vec<ValueRef<'s, T>> {
153		self.try_extract_sequence().expect("Failed to extract sequence")
154	}
155
156	#[inline]
157	pub fn len(&self) -> usize {
158		let mut len = 0;
159		ortsys![unsafe GetValueCount(self.ptr(), &mut len).expect("infallible")];
160		len
161	}
162
163	#[inline]
164	pub fn is_empty(&self) -> bool {
165		let mut len = 0;
166		ortsys![unsafe GetValueCount(self.ptr(), &mut len).expect("infallible")];
167		len == 0
168	}
169
170	pub fn get(&self, index: usize) -> Option<ValueRef<'_, T>> {
171		extract_from_sequence(self.ptr(), index, &Allocator::default())
172			.ok()
173			.and_then(|x| x.downcast().ok())
174	}

176	/// Converts from a strongly-typed [`Sequence<T>`] to a type-erased [`DynSequence`].

180	}

185		DynSequenceRef::new(Value {
186			inner: Arc::clone(&self.inner),
187			_markers: PhantomData
188		})
189	}

201fn extract_from_sequence<'s>(ptr: *const ort_sys::OrtValue, i: usize, allocator: &Allocator) -> Result<ValueRef<'s, DynValueTypeMarker>> {
202	let mut value_ptr = ptr::null_mut();
203	ortsys![unsafe GetValue(ptr, i as _, allocator.ptr().cast_mut(), &mut value_ptr)?; nonNull(value_ptr)];
204
205	let mut value = ValueRef::new(unsafe { Value::from_ptr(value_ptr, None) });
206	value.upgradable = false;
207	Ok(value)
