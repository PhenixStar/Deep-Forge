1//! ONNX Runtime doesn't (currently) expose an API for inter-device copies, so we instead use a dummy model to copy the

30/// A simple graph in `.ort` format with a single `Identity` node between the input & output.
31static IDENTITY_MODEL: &[u8] = include_bytes!("./identity.ort");
32static SESSIONS: OnceLock<Mutex<MiniMap<IdentitySessionKey, IdentitySession>>> = OnceLock::new();
33/// `RunOptions` with [`RunOptions::disable_device_sync`], shared across `to_async()` calls to reduce allocations.
34static IDENTITY_RUN_OPTIONS: OnceLock<RunOptions<NoSelectedOutputs>> = OnceLock::new();
35
36fn ep_for_device(device: AllocationDevice, device_id: i32) -> Result<ep::ExecutionProviderDispatch> {
37	Ok(match device {
38		AllocationDevice::CPU => ep::CPU::default().with_arena_allocator(false).build(),
39		AllocationDevice::CUDA | AllocationDevice::CUDA_PINNED => ep::CUDA::default()
40			.with_device_id(device_id)
41			.with_arena_extend_strategy(ep::ArenaExtendStrategy::SameAsRequested)
42			.with_conv_max_workspace(false)
43			.with_conv_algorithm_search(ep::cuda::ConvAlgorithmSearch::Default)
44			.build(),
45		AllocationDevice::DIRECTML => ep::DirectML::default().with_device_id(device_id).build(),
46		AllocationDevice::CANN | AllocationDevice::CANN_PINNED => ep::CANN::default()
47			.with_arena_extend_strategy(ep::ArenaExtendStrategy::SameAsRequested)
48			.with_cann_graph(false)
49			.with_device_id(device_id)
50			.build(),
51		AllocationDevice::OPENVINO_CPU | AllocationDevice::OPENVINO_GPU => ep::OpenVINO::default()
52			.with_num_threads(1)
53			.with_device_type(if device == AllocationDevice::OPENVINO_CPU {
54				"CPU".to_string()
55			} else {
56				format!("GPU.{device_id}")
57			})
58			.build(),
59		AllocationDevice::HIP | AllocationDevice::HIP_PINNED => ep::ROCm::default()
60			.with_arena_extend_strategy(ep::ArenaExtendStrategy::SameAsRequested)
61			.with_hip_graph(false)
62			.with_exhaustive_conv_search(false)
63			.with_device_id(device_id)
64			.build(),
65		_ => return Err(crate::Error::new("Unsupported allocation device {device} for tensor copy target"))
66	})
67}

69impl<Type: DefiniteTensorValueTypeMarker + ?Sized> Value<Type> {
70	/// Copies the contents of this tensor to another device, returning the newly created tensor value.
71	///
72	/// ```
73	/// # use ort::{memory::{Allocator, AllocatorType, AllocationDevice, MemoryInfo, MemoryType}, session::Session, value::Tensor};
74	/// # fn main() -> ort::Result<()> {
75	/// # if false {
76	/// # let session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
77	/// let cuda_allocator = Allocator::new(
78	/// 	&session,
79	/// 	MemoryInfo::new(AllocationDevice::CUDA, 0, AllocatorType::Device, MemoryType::Default)?
80	/// )?;
81	/// let cuda_tensor = Tensor::<f32>::new(&cuda_allocator, [1_usize, 3, 224, 224])?;
82	/// # }
83	/// # let cuda_tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;
84	///
85	/// let cpu_tensor = cuda_tensor.to(AllocationDevice::CPU, 0)?;
86	/// assert_eq!(cpu_tensor.memory_info().allocation_device(), AllocationDevice::CPU);
87	/// assert_eq!(**cpu_tensor.shape(), [1, 3, 224, 224]);
88	/// # Ok(())
89	/// # }
90	/// ```
91	#[cfg_attr(docsrs, doc(cfg(not(target_arch = "wasm32"))))]
92	pub fn to(&self, device: AllocationDevice, device_id: i32) -> Result<Value<Type>> {
93		self.copy_to_inner(device, device_id, |identity_session| {
94			let target_memory_info = MemoryInfo::new(device, device_id, AllocatorType::Device, MemoryType::Default)?;
95			identity_session.binding.bind_output_to_device("output", &target_memory_info)?;
96
97			let output = identity_session
98				.session
99				.run_binding(&identity_session.binding)?
100				.remove("output")
101				.expect("identity model should have single output");
102			Ok(unsafe { output.transmute_type() })
103		})
104	}
105
106	/// Asynchronously copies the contents of this tensor to another device.
107	///
108	/// Unlike [`Tensor::to`][crate::value::Tensor::to], the device's stream will *not* be synchronized (like via
109	/// `cudaStreamSynchronize`); thus this function is most useful for host-to-device transfers.
110	///
111	/// ```
112	/// # use ort::{memory::{Allocator, AllocatorType, AllocationDevice, MemoryInfo, MemoryType}, session::Session, value::Tensor};
113	/// # fn main() -> ort::Result<()> {
114	/// # let tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;
115	/// # if false {
116	/// let cuda_tensor = tensor.to_async(AllocationDevice::CUDA, 0)?;
117	/// // pass to other CUDA code, or to session input
118	/// # }
119	/// # Ok(())
120	/// # }
121	/// ```
122	#[cfg_attr(docsrs, doc(cfg(not(target_arch = "wasm32"))))]
123	pub fn to_async(&self, device: AllocationDevice, device_id: i32) -> Result<Value<Type>> {
124		self.copy_to_inner(device, device_id, |identity_session| {
125			let target_memory_info = MemoryInfo::new(device, device_id, AllocatorType::Device, MemoryType::Default)?;
126			identity_session.binding.bind_output_to_device("output", &target_memory_info)?;
127
128			let options = IDENTITY_RUN_OPTIONS.get_or_try_init(|| -> Result<RunOptions> {
129				let mut options = RunOptions::new()?;
130				options.disable_device_sync()?;
131				Ok(options)
132			})?;
133			let output = identity_session
134				.session
135				.run_binding_with_options(&identity_session.binding, options)?
136				.remove("output")
137				.expect("identity model should have single output");
138			Ok(unsafe { output.transmute_type() })
139		})
140	}
141
142	/// Copies the contents of this tensor to another tensor potentially residing on a separate device.
143	///
144	/// ```
145	/// # use ort::{memory::{Allocator, AllocatorType, AllocationDevice, MemoryInfo, MemoryType}, session::Session, value::Tensor};
146	/// # fn main() -> ort::Result<()> {
147	/// # if false {
148	/// # let session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
149	/// let cuda_allocator = Allocator::new(
150	/// 	&session,
151	/// 	MemoryInfo::new(AllocationDevice::CUDA, 0, AllocatorType::Device, MemoryType::Default)?
152	/// )?;
153	/// let cuda_tensor = Tensor::<f32>::new(&cuda_allocator, [1_usize, 3, 224, 224])?;
154	/// # }
155	/// # let cuda_tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;
156	/// let mut cpu_tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;;
157	///
158	/// cuda_tensor.copy_into(&mut cpu_tensor)?;
159	/// # Ok(())
160	/// # }
161	/// ```
162	#[cfg_attr(docsrs, doc(cfg(not(target_arch = "wasm32"))))]
163	pub fn copy_into(&self, target: &mut Value<Type>) -> Result<()> {
164		if self.dtype() != target.dtype() {
165			return Err(Error::new("target data type does not match source data type"));
166		} else if self.shape() != target.shape() {
167			return Err(Error::new("target shape does not match source shape"));
168		}
169
170		let target_memory_info = target.memory_info();
171		self.copy_to_inner(target_memory_info.allocation_device(), target_memory_info.device_id(), |identity_session| {
172			unsafe { identity_session.binding.bind_output_mut("output", target) }?;
173			identity_session.session.run_binding(&identity_session.binding)?;
174			Ok(())
175		})
176	}

178	/// Asynchronously copies the contents of this tensor to another tensor.
179	///
180	/// Unlike [`Tensor::copy_into`][crate::value::Tensor::copy_into], the device's stream will *not* be synchronized
181	/// (like via `cudaStreamSynchronize`); thus this function is most useful for host-to-device transfers.
182	///
183	/// ```
184	/// # use ort::{memory::{Allocator, AllocatorType, AllocationDevice, MemoryInfo, MemoryType}, session::Session, value::Tensor};
185	/// # fn main() -> ort::Result<()> {
186	/// let cpu_tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;;
187	/// # if false {
188	/// # let session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
189	/// let cuda_allocator = Allocator::new(
190	/// 	&session,
191	/// 	MemoryInfo::new(AllocationDevice::CUDA, 0, AllocatorType::Device, MemoryType::Default)?
192	/// )?;
193	/// let mut cuda_tensor = Tensor::<f32>::new(&cuda_allocator, [1_usize, 3, 224, 224])?;
194	/// # }
195	/// # let mut cuda_tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;
196	///
197	/// cpu_tensor.copy_into_async(&mut cuda_tensor)?;
198	/// # Ok(())
199	/// # }
200	/// ```
201	#[cfg_attr(docsrs, doc(cfg(not(target_arch = "wasm32"))))]
202	pub fn copy_into_async(&self, target: &mut Value<Type>) -> Result<()> {
203		if self.dtype() != target.dtype() {
204			return Err(Error::new("target data type does not match source data type"));
205		} else if self.shape() != target.shape() {
206			return Err(Error::new("target shape does not match source shape"));
207		}
208
209		let target_memory_info = target.memory_info();
210		self.copy_to_inner(target_memory_info.allocation_device(), target_memory_info.device_id(), |identity_session| {
211			unsafe { identity_session.binding.bind_output_mut("output", target) }?;
212			let options = IDENTITY_RUN_OPTIONS.get_or_try_init(|| -> Result<RunOptions> {
213				let mut options = RunOptions::new()?;
214				options.disable_device_sync()?;
215				Ok(options)
216			})?;
217			identity_session.session.run_binding_with_options(&identity_session.binding, options)?;
218			Ok(())
219		})
220	}
221
222	fn copy_to_inner<F, T>(&self, device: AllocationDevice, device_id: i32, runner: F) -> Result<T>
223	where
224		F: FnOnce(&mut IdentitySession) -> Result<T>
225	{
226		let source_memory_info = self.memory_info();
227		let tensor_type = ort_sys::ONNXTensorElementDataType::from(*self.data_type());
228
229		let mut identity_session = IdentitySessionHandle::new(source_memory_info, device, device_id, tensor_type)?;
230		identity_session.binding.bind_input("input", self)?;
231		let res = runner(&mut identity_session);
232		identity_session.binding.clear();
233		res
234	}
235}

237#[cfg_attr(docsrs, doc(cfg(not(target_arch = "wasm32"))))]
238impl<Type: DefiniteTensorValueTypeMarker + ?Sized> Clone for Value<Type> {
239	/// Creates a copy of this tensor and its data on the same device it resides on.
240	///
241	/// ```
242	/// # use ort::{value::Tensor, AsPointer};
243	/// # fn main() -> ort::Result<()> {
244	/// let array = vec![1_i64, 2, 3, 4, 5];
245	/// let tensor = Tensor::from_array(([array.len()], array.into_boxed_slice()))?;
246	///
247	/// let new_tensor = tensor.clone();
248	///
249	/// // same data
250	/// assert_eq!(tensor.extract_tensor(), new_tensor.extract_tensor());
251	///
252	/// // different allocations
253	/// assert_ne!(tensor.ptr(), new_tensor.ptr());
254	/// assert_ne!(tensor.data_ptr(), new_tensor.data_ptr());
255	/// # 	Ok(())
256	/// # }
257	/// ```
258	#[cfg_attr(docsrs, doc(cfg(not(target_arch = "wasm32"))))]
259	fn clone(&self) -> Self {
260		let memory_info = self.memory_info();
261		self.to(memory_info.allocation_device(), memory_info.device_id())
262			.expect("Failed to clone tensor")
263	}
264}

266struct IdentitySessionHandle {
267	inner: &'static mut IdentitySession,
268	_guard: MutexGuard<'static, MiniMap<IdentitySessionKey, IdentitySession>>
269}
270
271impl IdentitySessionHandle {
272	fn new(
273		source_memory_info: &MemoryInfo,
274		target_device: AllocationDevice,
275		target_device_id: i32,
276		tensor_type: ort_sys::ONNXTensorElementDataType
277	) -> Result<Self> {
278		let session_key = IdentitySessionKey {
279			src_device: source_memory_info.allocation_device(),
280			src_device_id: source_memory_info.device_id(),
281			target_device,
282			target_device_id,
283			dtype: tensor_type
284		};
285		let mut sessions = SESSIONS.get_or_init(|| Mutex::new(MiniMap::new())).lock();
286		let identity_session = match sessions.get_mut(&session_key) {
287			Some(entry) => entry,
288			None => {
289				let mut model_bytes = IDENTITY_MODEL.to_vec();
290				// Override the expected element type of the input & output nodes, respectively.
291				model_bytes[544] = tensor_type as u8;
292				model_bytes[604] = tensor_type as u8;
293
294				let (source_ep, target_ep) = (
295					// We enable `.error_on_failure()` here since `IoBinding::bind_output_to_device` will silently fall back to binding to CPU if the target
296					// device doesn't have an EP registered.
297					ep_for_device(source_memory_info.allocation_device(), source_memory_info.device_id())?.error_on_failure(),
298					ep_for_device(target_device, target_device_id)?.error_on_failure()
299				);
300
301				let mut builder = Session::builder()?
302					.with_optimization_level(GraphOptimizationLevel::Level3)?
303					// since these sessions are persistent for the lifetime of the program, keep them as lean as possible
304					// by disabling threading & memory optimizations (there's only 1 operation in the graph anyway)
305					.with_intra_threads(1)?
306					.with_inter_threads(1)?
307					.with_inter_op_spinning(false)?
308					.with_intra_op_spinning(false)?
309					.with_memory_pattern(false)?
310					.with_no_environment_execution_providers()?;
311				// Avoid registering the same EP twice, since that's an error.
312				if source_ep.inner.name() != target_ep.inner.name() {
313					builder = builder.with_execution_providers([source_ep, target_ep])?;
314				} else {
315					builder = builder.with_execution_providers([source_ep])?;
316				}
317
318				let session = builder.commit_from_memory(&model_bytes)?;
319				let binding = session.create_binding()?;
320				sessions.insert(session_key.clone(), IdentitySession { session, binding });
321				sessions.get_mut(&session_key).expect("insert should have worked")
322			}
323		};
324
325		Ok(Self {
326			inner: unsafe { core::mem::transmute::<&mut IdentitySession, &'static mut IdentitySession>(identity_session) },
327			_guard: sessions
328		})
329	}
330}
