19/// The type of a [`Value`][super::Value], or a session input/output.
20///
21/// ```
22/// # use std::sync::Arc;
23/// # use ort::{session::Session, value::{ValueType, Tensor, Shape, SymbolicDimensions, TensorElementType}};
24/// # fn main() -> ort::Result<()> {
25/// # 	let session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
26/// // `ValueType`s can be obtained from session inputs/outputs:
27/// let input = &session.inputs()[0];
28/// assert_eq!(
29/// 	input.dtype(),
30/// 	&ValueType::Tensor {
31/// 		ty: TensorElementType::Float32,
32/// 		// Our model's input has 3 dynamic dimensions, represented by -1
33/// 		shape: Shape::new([-1, -1, -1, 3]),
34/// 		// Dynamic dimensions may also have names.
35/// 		dimension_symbols: SymbolicDimensions::new([
36/// 			"unk__31".to_string(),
37/// 			"unk__32".to_string(),
38/// 			"unk__33".to_string(),
39/// 			String::default()
40/// 		])
41/// 	}
42/// );
43///
44/// // ...or by `Value`s created in Rust or output by a session.
45/// let value = Tensor::from_array(([5usize], vec![1_i64, 2, 3, 4, 5].into_boxed_slice()))?;
46/// assert_eq!(
47/// 	value.dtype(),
48/// 	&ValueType::Tensor {
49/// 		ty: TensorElementType::Int64,
50/// 		shape: Shape::new([5]),
51/// 		dimension_symbols: SymbolicDimensions::new([String::default()])
52/// 	}
53/// );
54/// # 	Ok(())
55/// # }
56/// ```
57#[derive(Debug, PartialEq, Eq, Clone)]
58pub enum ValueType {
59	/// Value is a tensor/multi-dimensional array.
60	Tensor {
61		/// Element type of the tensor.
62		ty: TensorElementType,
63		/// Shape of the tensor. If an exact dimension is not known (i.e. a dynamic dimension as part of an
64		/// [`Outlet`]), the dimension will be `-1`.
65		///
66		/// Actual tensor values (i.e. not [`Outlet`] definitions), which have a known dimension, will
67		/// always have non-negative dimensions.
68		shape: Shape,
69		dimension_symbols: SymbolicDimensions
70	},
71	/// A sequence (vector) of other `Value`s.
72	///
73	/// [Per ONNX spec](https://onnx.ai/onnx/intro/concepts.html#other-types), only sequences of tensors and maps are allowed.
74	Sequence(Box<ValueType>),
75	/// A map/dictionary from one element type to another.
76	Map {
77		/// The map key type. Allowed types are:
78		/// - [`TensorElementType::Int8`]
79		/// - [`TensorElementType::Int16`]
80		/// - [`TensorElementType::Int32`]
81		/// - [`TensorElementType::Int64`]
82		/// - [`TensorElementType::Uint8`]
83		/// - [`TensorElementType::Uint16`]
84		/// - [`TensorElementType::Uint32`]
85		/// - [`TensorElementType::Uint64`]
86		/// - [`TensorElementType::String`]
87		key: TensorElementType,
88		/// The map value type.
89		value: TensorElementType
90	},
91	/// An optional value, which may or may not contain a [`Value`][super::Value].
92	Optional(Box<ValueType>)
93}

96	pub(crate) unsafe fn from_type_info(typeinfo_ptr: NonNull<ort_sys::OrtTypeInfo>) -> Self {
97		let _guard = util::run_on_drop(|| {
98			ortsys![unsafe ReleaseTypeInfo(typeinfo_ptr.as_ptr())];
99		});
100
101		let mut ty: ort_sys::ONNXType = ort_sys::ONNXType::ONNX_TYPE_UNKNOWN;
102		ortsys![unsafe GetOnnxTypeFromTypeInfo(typeinfo_ptr.as_ptr(), &mut ty).expect("infallible")];
103		match ty {
104			ort_sys::ONNXType::ONNX_TYPE_TENSOR | ort_sys::ONNXType::ONNX_TYPE_SPARSETENSOR => {
105				let mut info_ptr: *const ort_sys::OrtTensorTypeAndShapeInfo = ptr::null_mut();
106				ortsys![unsafe CastTypeInfoToTensorInfo(typeinfo_ptr.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
107				unsafe { extract_data_type_from_tensor_info(info_ptr) }
108			}
109			ort_sys::ONNXType::ONNX_TYPE_SEQUENCE => {
110				let mut info_ptr: *const ort_sys::OrtSequenceTypeInfo = ptr::null_mut();
111				ortsys![unsafe CastTypeInfoToSequenceTypeInfo(typeinfo_ptr.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
112
113				let mut element_type_info: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
114				ortsys![unsafe GetSequenceElementType(info_ptr.as_ptr(), &mut element_type_info).expect("infallible"); nonNull(element_type_info)];
115				let _guard = util::run_on_drop(|| {
116					ortsys![unsafe ReleaseTypeInfo(element_type_info.as_ptr())];
117				});
118
119				let mut ty: ort_sys::ONNXType = ort_sys::ONNXType::ONNX_TYPE_UNKNOWN;
120				ortsys![unsafe GetOnnxTypeFromTypeInfo(element_type_info.as_ptr(), &mut ty).expect("infallible")];
121
122				match ty {
123					ort_sys::ONNXType::ONNX_TYPE_TENSOR => {
124						let mut info_ptr: *const ort_sys::OrtTensorTypeAndShapeInfo = ptr::null_mut();
125						ortsys![unsafe CastTypeInfoToTensorInfo(element_type_info.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
126						let ty = unsafe { extract_data_type_from_tensor_info(info_ptr) };
127						ValueType::Sequence(Box::new(ty))
128					}
129					ort_sys::ONNXType::ONNX_TYPE_MAP => {
130						let mut info_ptr: *const ort_sys::OrtMapTypeInfo = ptr::null_mut();
131						ortsys![unsafe CastTypeInfoToMapTypeInfo(element_type_info.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
132						let ty = unsafe { extract_data_type_from_map_info(info_ptr) };
133						ValueType::Sequence(Box::new(ty))
134					}
135					_ => unreachable!()
136				}
137			}
138			ort_sys::ONNXType::ONNX_TYPE_MAP => {
139				let mut info_ptr: *const ort_sys::OrtMapTypeInfo = ptr::null_mut();
140				ortsys![unsafe CastTypeInfoToMapTypeInfo(typeinfo_ptr.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
141				unsafe { extract_data_type_from_map_info(info_ptr) }
142			}
143			ort_sys::ONNXType::ONNX_TYPE_OPTIONAL => {
144				let mut info_ptr: *const ort_sys::OrtOptionalTypeInfo = ptr::null_mut();
145				ortsys![unsafe CastTypeInfoToOptionalTypeInfo(typeinfo_ptr.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
146
147				let mut contained_type: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
148				ortsys![unsafe GetOptionalContainedTypeInfo(info_ptr.as_ptr(), &mut contained_type).expect("infallible"); nonNull(contained_type)];
149
150				ValueType::Optional(Box::new(unsafe { ValueType::from_type_info(contained_type) }))
151			}
152			_ => unreachable!()
153		}
154	}

156	pub(crate) fn to_tensor_type_info(&self) -> Option<*mut ort_sys::OrtTensorTypeAndShapeInfo> {
157		match self {
158			Self::Tensor { ty, shape, dimension_symbols } => {
159				let mut info_ptr = ptr::null_mut();
160				ortsys![unsafe CreateTensorTypeAndShapeInfo(&mut info_ptr).expect("infallible")];
161				ortsys![unsafe SetTensorElementType(info_ptr, (*ty).into()).expect("infallible")];
162				ortsys![unsafe SetDimensions(info_ptr, shape.as_ptr(), shape.len()).expect("infallible")];
163				with_cstr_ptr_array(dimension_symbols, &|ptrs| {
164					ortsys![unsafe SetSymbolicDimensions(info_ptr, ptrs.as_ptr().cast_mut(), dimension_symbols.len()).expect("infallible")];
165					Ok(())
166				})
167				.expect("invalid dimension symbols");
168				Some(info_ptr)
169			}
170			_ => None
171		}
172	}

174	/// Converts this type to an [`ort_sys::OrtTypeInfo`] using the Model Editor API, so it shouldn't be used outside of
175	/// `crate::editor`
176	#[cfg(feature = "api-22")]
177	pub(crate) fn to_type_info(&self) -> Result<*mut ort_sys::OrtTypeInfo> {
178		let mut info_ptr: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
179		match self {
180			Self::Tensor { .. } => {
181				let tensor_type_info = self.to_tensor_type_info().expect("infallible");
182				let _guard = util::run_on_drop(|| ortsys![unsafe ReleaseTensorTypeAndShapeInfo(tensor_type_info)]);
183				ortsys![@editor: unsafe CreateTensorTypeInfo(tensor_type_info, &mut info_ptr)?];
184			}
185			Self::Map { .. } => {
186				todo!();
187			}
188			Self::Sequence(ty) => {
189				let el_type = ty.to_type_info()?;
190				let _guard = util::run_on_drop(|| ortsys![unsafe ReleaseTypeInfo(el_type)]);
191				ortsys![@editor: unsafe CreateSequenceTypeInfo(el_type, &mut info_ptr)?];
192			}
193			Self::Optional(ty) => {
194				let ty = ty.to_type_info()?;
195				let _guard = util::run_on_drop(|| ortsys![unsafe ReleaseTypeInfo(ty)]);
196				ortsys![@editor: unsafe CreateOptionalTypeInfo(ty, &mut info_ptr)?];
197			}
198		}
199		Ok(info_ptr)
200	}

262impl fmt::Display for ValueType {
263	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
264		match self {
265			ValueType::Tensor { ty, shape, dimension_symbols } => {
266				write!(f, "Tensor<{ty}>(")?;
267				for (i, dimension) in shape.iter().copied().enumerate() {
268					if dimension == -1 {
269						let sym = &dimension_symbols[i];
270						if sym.is_empty() {
271							f.write_str("dyn")?;
272						} else {
273							f.write_str(sym)?;
274						}
275					} else {
276						dimension.fmt(f)?;
277					}
278					if i != shape.len() - 1 {
279						f.write_str(", ")?;
280					}
281				}
282				f.write_str(")")?;
283				Ok(())
284			}
285			ValueType::Map { key, value } => write!(f, "Map<{key}, {value}>"),
286			ValueType::Sequence(inner) => write!(f, "Sequence<{inner}>"),
287			ValueType::Optional(inner) => write!(f, "Option<{inner}>")
288		}
289	}

292/// The input/output to a [session](crate::session::Session), [node](crate::editor::Node), or
293/// [operator](crate::operator::Operator), consisting of a name and expected [data type](ValueType).
294#[derive(Debug)]
295pub struct Outlet {
296	name: String,
297	dtype: ValueType,
298	// Outlet is used for many things, but a ValueInfo can only be created if the Model Editor API is available, which it sometimes may not be.
299	value_info: Option<NonNull<ort_sys::OrtValueInfo>>,
300	drop: bool
301}

303impl Outlet {
304	pub fn new<S: Into<String>>(name: S, dtype: ValueType) -> Self {
305		let name = name.into();
306		let value_info = Self::make_value_info(&name, &dtype);
307		Self {
308			name,
309			dtype,
310			value_info,
311			drop: value_info.is_some()
312		}
313	}
314
315	#[cfg(feature = "api-22")]
316	pub(crate) unsafe fn from_raw(raw: NonNull<ort_sys::OrtValueInfo>, drop: bool) -> Result<Self> {
317		let mut name = ptr::null();
318		ortsys![unsafe GetValueInfoName(raw.as_ptr(), &mut name)?];
319		let name = if !name.is_null() {
320			unsafe { CStr::from_ptr(name) }.to_str().map_or_else(|_| String::new(), str::to_string)
321		} else {
322			String::new()
323		};
324
325		let mut type_info = ptr::null();
326		ortsys![unsafe GetValueInfoTypeInfo(raw.as_ptr(), &mut type_info)?; nonNull(type_info)];
327		let dtype = unsafe { ValueType::from_type_info(type_info) };
328
329		Ok(Self {
330			name,
331			dtype,
332			value_info: Some(raw),
333			drop
334		})
335	}

348	pub(crate) fn make_value_info(name: &str, dtype: &ValueType) -> Option<NonNull<ort_sys::OrtValueInfo>> {
349		let type_info = dtype.to_type_info().ok()?;
350		let _guard = run_on_drop(|| ortsys![unsafe ReleaseTypeInfo(type_info)]);
351
352		with_cstr(name.as_bytes(), &|name| {
353			let mut ptr: *mut ort_sys::OrtValueInfo = ptr::null_mut();
354			ortsys![@editor: unsafe CreateValueInfo(name.as_ptr(), type_info, &mut ptr)?; nonNull(ptr)];
355			Ok(ptr)
356		})
357		.ok()
358	}

382pub(crate) unsafe fn extract_data_type_from_tensor_info(info_ptr: NonNull<ort_sys::OrtTensorTypeAndShapeInfo>) -> ValueType {
383	let mut type_sys = ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED;
384	ortsys![unsafe GetTensorElementType(info_ptr.as_ptr(), &mut type_sys).expect("infallible")];
385	assert_ne!(type_sys, ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED);
386	// This transmute should be safe since its value is read from GetTensorElementType, which we must trust
387	let mut num_dims = 0;
388	ortsys![unsafe GetDimensionsCount(info_ptr.as_ptr(), &mut num_dims).expect("infallible")];
389
390	let mut node_dims = Shape::empty(num_dims);
391	ortsys![unsafe GetDimensions(info_ptr.as_ptr(), node_dims.as_mut_ptr(), num_dims).expect("infallible")];
392
393	let mut symbolic_dims: SmallVec<[_; 4]> = smallvec![ptr::null(); num_dims];
394	ortsys![unsafe GetSymbolicDimensions(info_ptr.as_ptr(), symbolic_dims.as_mut_ptr(), num_dims).expect("infallible")];
395
396	let dimension_symbols = symbolic_dims
397		.into_iter()
398		.map(|c| unsafe { CStr::from_ptr(c) }.to_str().map_or_else(|_| String::new(), str::to_string))
399		.collect();
400
401	ValueType::Tensor {
402		ty: type_sys.into(),
403		shape: node_dims,
404		dimension_symbols
405	}
406}

408unsafe fn extract_data_type_from_map_info(info_ptr: NonNull<ort_sys::OrtMapTypeInfo>) -> ValueType {
409	let mut key_type_sys = ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED;
410	ortsys![unsafe GetMapKeyType(info_ptr.as_ptr(), &mut key_type_sys).expect("infallible")];
411	assert_ne!(key_type_sys, ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED);
412
413	let mut value_type_info: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
414	ortsys![unsafe GetMapValueType(info_ptr.as_ptr(), &mut value_type_info).expect("infallible")];
415	let _guard = util::run_on_drop(|| {
416		ortsys![unsafe ReleaseTypeInfo(value_type_info)];
417	});
418
419	let mut value_info_ptr: *const ort_sys::OrtTensorTypeAndShapeInfo = ptr::null_mut();
420	ortsys![unsafe CastTypeInfoToTensorInfo(value_type_info, &mut value_info_ptr).expect("infallible")];
421	let mut value_type_sys = ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED;
422	ortsys![unsafe GetTensorElementType(value_info_ptr, &mut value_type_sys).expect("infallible")];
423	assert_ne!(value_type_sys, ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED);
424
425	ValueType::Map {
426		key: key_type_sys.into(),
427		value: value_type_sys.into()
428	}
429}
