37/// `Result` type returned by [`SessionBuilder`] methods.
38///
39/// This type supports [error recovery](Error::recover):
40/// ```
41/// # use ort::session::{builder::GraphOptimizationLevel, Session};
42/// # fn main() -> ort::Result<()> {
43/// let session = Session::builder()?
44/// 	.with_optimization_level(GraphOptimizationLevel::All)
45/// 	// Optimization isn't enabled in minimal builds of ONNX Runtime, so throws an error. We can just ignore it.
46/// 	.unwrap_or_else(|e| e.recover())
47/// 	.commit_from_file("tests/data/upsample.onnx")?;
48/// # Ok(())
49/// # }
50/// ```
51pub type BuilderResult = Result<SessionBuilder, Error<SessionBuilder>>;

53/// Creates a session using the builder pattern.
54///
55/// Once configured, use the
56/// [`SessionBuilder::commit_from_file`](crate::session::builder::SessionBuilder::commit_from_file) method to 'commit'
57/// the builder configuration into a [`Session`].
58///
59/// ```
60/// # use ort::session::{builder::GraphOptimizationLevel, Session};
61/// # fn main() -> ort::Result<()> {
62/// let session = Session::builder()?
63/// 	.with_optimization_level(GraphOptimizationLevel::Level1)?
64/// 	.with_intra_threads(1)?
65/// 	.commit_from_file("tests/data/upsample.onnx")?;
66/// # Ok(())
67/// # }
68/// ```
69///
70/// [`Session`]: crate::session::Session
71pub struct SessionBuilder {
72	session_options_ptr: Arc<SessionOptionsPointer>,
73	memory_info: Option<Arc<MemoryInfo>>,
74	operator_domains: SmallVec<[Arc<OperatorDomain>; 1]>,
75	initializers: Vec<Arc<DynValue>>,
76	external_initializer_buffers: Vec<Cow<'static, [u8]>>,
77	prepacked_weights: Option<PrepackedWeights>,
78	thread_manager: Option<Arc<dyn Any>>,
79	logger: Option<Arc<LoggerFunction>>,
80	no_global_thread_pool: bool,
81	no_env_eps: bool,
82	pub(crate) environment: Arc<Environment>
83}

109impl SessionBuilder {
110	/// Creates a new session builder.
111	///
112	/// ```
113	/// # use ort::session::{builder::GraphOptimizationLevel, Session};
114	/// # fn main() -> ort::Result<()> {
115	/// let session = Session::builder()?
116	/// 	.with_optimization_level(GraphOptimizationLevel::Level1)?
117	/// 	.with_intra_threads(1)?
118	/// 	.commit_from_file("tests/data/upsample.onnx")?;
119	/// # Ok(())
120	/// # }
121	/// ```
122	pub fn new() -> Result<Self> {
123		let environment = environment::current()?;
124
125		let mut session_options_ptr: *mut ort_sys::OrtSessionOptions = ptr::null_mut();
126		ortsys![unsafe CreateSessionOptions(&mut session_options_ptr)?; nonNull(session_options_ptr)];
127
128		// target on-device usage; prefer efficiency by default
129		// .with_execution_providers/.with_auto_ep will override this
130		#[cfg(feature = "api-22")]
131		let _ = ortsys![@ort: unsafe SessionOptionsSetEpSelectionPolicy(session_options_ptr.as_ptr(), AutoDevicePolicy::MaxEfficiency.into()) as Result];
132
133		Ok(Self {
134			session_options_ptr: Arc::new(SessionOptionsPointer::new(session_options_ptr)),
135			memory_info: None,
136			operator_domains: SmallVec::new(),
137			initializers: Vec::new(),
138			external_initializer_buffers: Vec::new(),
139			prepacked_weights: None,
140			thread_manager: None,
141			logger: None,
142			no_global_thread_pool: false,
143			no_env_eps: false,
144			environment
145		})
146	}

148	#[inline]
149	pub(crate) fn add_config_entry(&mut self, key: impl AsRef<str>, value: impl AsRef<str>) -> Result<()> {
150		let ptr = self.ptr_mut();
151		with_cstr(key.as_ref().as_bytes(), &|key| {
152			with_cstr(value.as_ref().as_bytes(), &|value| {
153				ortsys![unsafe AddSessionConfigEntry(ptr, key.as_ptr(), value.as_ptr())?];
154				Ok(())
155			})
156		})
157	}

159	/// Creates a signaler that can be used from another thread to cancel any in-progress commits.
160	///
161	/// ```
162	/// # use ort::session::{builder::GraphOptimizationLevel, Session};
163	/// # use std::{thread, time::Duration};
164	/// # fn main() -> ort::Result<()> {
165	/// let mut builder = Session::builder()?
166	/// 	.with_optimization_level(GraphOptimizationLevel::Level1)?
167	/// 	.with_intra_threads(1)?;
168	///
169	/// let canceler = builder.canceler();
170	/// thread::spawn(move || {
171	/// 	thread::sleep(Duration::from_millis(500));
172	/// 	// timeout if model hasn't loaded in 500ms
173	/// 	let _ = canceler.cancel();
174	/// });
175	///
176	/// let session = builder.commit_from_file("tests/data/upsample.onnx")?;
177	/// # Ok(())
178	/// # }
179	/// ```
180	#[cfg(feature = "api-22")]
181	#[cfg_attr(docsrs, doc(cfg(feature = "api-22")))]
182	pub fn canceler(&self) -> LoadCanceler {
183		LoadCanceler(Arc::downgrade(&self.session_options_ptr))
184	}
185
186	/// Adds a custom configuration entry to the session.
187	pub fn with_config_entry(mut self, key: impl AsRef<str>, value: impl AsRef<str>) -> BuilderResult {
188		match self.add_config_entry(key.as_ref(), value.as_ref()) {
189			Ok(()) => Ok(self),
190			Err(e) => Err(e.with_recover(self))
191		}
192	}
193}

203/// A handle which can be used to remotely terminate an in-progress session load.
204///
205/// See [`SessionBuilder::canceler`].
206#[derive(Debug, Clone)]
207#[cfg(feature = "api-22")]
208#[cfg_attr(docsrs, doc(cfg(feature = "api-22")))]
209pub struct LoadCanceler(Weak<SessionOptionsPointer>);
210
211#[cfg(feature = "api-22")]
212unsafe impl Send for LoadCanceler {}
213#[cfg(feature = "api-22")]
214unsafe impl Sync for LoadCanceler {}
215
216#[cfg(feature = "api-22")]
217impl LoadCanceler {
218	/// Cancels any active session commits.
219	///
220	/// ```
221	/// # use ort::session::{builder::GraphOptimizationLevel, Session};
222	/// # use std::{thread, time::Duration};
223	/// # fn main() -> ort::Result<()> {
224	/// let mut builder = Session::builder()?
225	/// 	.with_optimization_level(GraphOptimizationLevel::Level1)?
226	/// 	.with_intra_threads(1)?;
227	///
228	/// let canceler = builder.canceler();
229	/// thread::spawn(move || {
230	/// 	thread::sleep(Duration::from_millis(500));
231	/// 	// timeout if model hasn't loaded in 500ms
232	/// 	let _ = canceler.cancel();
233	/// });
234	///
235	/// let session = builder.commit_from_file("tests/data/upsample.onnx")?;
236	/// # Ok(())
237	/// # }
238	/// ```
239	#[cfg(feature = "api-22")]
240	#[cfg_attr(docsrs, doc(cfg(feature = "api-22")))]
241	pub fn cancel(&self) -> Result<()> {
242		if let Some(ptr) = self.0.upgrade() {
243			ortsys![unsafe SessionOptionsSetLoadCancellationFlag(ptr.as_ptr(), true)?];
244		}
245		Ok(())
246	}
247}
