28/// An input adapter, allowing for loading many static inputs from disk at once.
29///
30/// [`Adapter`] essentially acts as a collection of predefined inputs allocated on a specific device that can easily be
31/// swapped out between session runs via [`RunOptions::add_adapter`]. With slight modifications to the session
32/// graph, [`Adapter`]s can be used as low-rank adapters (LoRAs) or as containers of style embeddings.
33///
34/// # Example
35/// An adapter can be created in Python with the `AdapterFormat` class:
36/// ```python
37/// import numpy as np
38/// import onnxruntime as ort
39///
40/// param_a = ort.OrtValue.ortvalue_from_numpy(np.array([[3], [4], [5], [6]], dtype=np.float32))
41/// param_b = ort.OrtValue.ortvalue_from_numpy(np.array([[7, 8, 9, 10]], dtype=np.float32))
42///
43/// adapter = ort.AdapterFormat()
44/// adapter.set_parameters({
45/// 	'lora_param_a': param_a,
46/// 	'lora_param_b': param_b
47/// })
48/// adapter.export_adapter('tests/data/adapter.orl')
49/// ```
50///
51/// Then, in Rust:
52/// ```
53/// # use ort::{session::{Adapter, RunOptions, Session}, value::Tensor};
54/// # fn main() -> ort::Result<()> {
55/// let mut model = Session::builder()?.commit_from_file("tests/data/lora_model.onnx")?;
56/// let lora = Adapter::from_file("tests/data/adapter.orl", None)?;
57///
58/// let mut run_options = RunOptions::new()?;
59/// run_options.add_adapter(&lora)?;
60///
61/// let outputs =
62/// 	model.run_with_options(ort::inputs![Tensor::<f32>::from_array(([4, 4], vec![1.0; 16]))?], &run_options)?;
63/// # Ok(())
64/// # }
65/// ```
66///
67/// Using [`Adapter`] is identical to, but more convenient than:
68/// ```
69/// # use ort::{session::{Adapter, RunOptions, Session}, value::Tensor};
70/// # fn main() -> ort::Result<()> {
71/// let mut model = Session::builder()?.commit_from_file("tests/data/lora_model.onnx")?;
72///
73/// // Load our parameters from disk somehow
74/// let param_a = Tensor::<f32>::from_array(([4, 1], vec![3., 4., 5., 6.]))?;
75/// let param_b = Tensor::<f32>::from_array(([1, 4], vec![7., 8., 6., 10.]))?;
76///
77/// let outputs = model.run(ort::inputs![
78/// 	"input" => Tensor::<f32>::from_array(([4, 4], vec![1.0; 16]))?,
79/// 	// Adapter parameters are just inputs.
80/// 	"lora_param_a" => param_a.view(),
81/// 	"lora_param_b" => param_b.view()
82/// ])?;
83/// # Ok(())
84/// # }
85/// ```
86///
87/// [`Adapter`] also lets us copy the parameters to a GPU at load time, so that they don't need to be copied on each
88/// session run.
89///
90/// [`RunOptions::add_adapter`]: crate::session::RunOptions::add_adapter
91#[cfg_attr(docsrs, doc(cfg(feature = "api-20")))]

97impl Adapter {
98	/// Loads an [`Adapter`] from a file.
99	///
100	/// An optional [`Allocator`] can be provided to specify the device on which the inputs should be allocated on.
101	/// Note that providing a CPU allocator will return an error; only device allocators are expected.
102	///
103	/// ```
104	/// # use ort::{
105	/// # 	ep,
106	/// # 	memory::DeviceType,
107	/// # 	session::{Adapter, RunOptions, Session},
108	/// # 	value::Tensor
109	/// # };
110	/// # fn main() -> ort::Result<()> {
111	/// let mut model = Session::builder()?
112	/// 	.with_execution_providers([ep::CUDA::default().build()])?
113	/// 	.commit_from_file("tests/data/lora_model.onnx")?;
114	///
115	/// let allocator = model.allocator();
116	/// let lora = Adapter::from_file(
117	/// 	"tests/data/adapter.orl",
118	/// 	if allocator.memory_info().device_type() == DeviceType::CPU { None } else { Some(&allocator) }
119	/// )?;
120	///
121	/// let mut run_options = RunOptions::new()?;
122	/// run_options.add_adapter(&lora)?;
123	///
124	/// let outputs =
125	/// 	model.run_with_options(ort::inputs![Tensor::<f32>::from_array(([4, 4], vec![1.0; 16]))?], &run_options)?;
126	/// # Ok(())
127	/// # }
128	/// ```
129	#[cfg(feature = "std")]
130	#[cfg_attr(docsrs, doc(cfg(feature = "std")))]
131	pub fn from_file(path: impl AsRef<Path>, allocator: Option<&Allocator>) -> Result<Self> {
132		let path = crate::util::path_to_os_char(path);
133		let allocator_ptr = allocator.map(|c| c.ptr().cast_mut()).unwrap_or_else(ptr::null_mut);
134		let mut ptr = ptr::null_mut();
135		ortsys![unsafe CreateLoraAdapter(path.as_ptr(), allocator_ptr, &mut ptr)?; nonNull(ptr)];
136		crate::logging::create!(Adapter, ptr);
137		Ok(Adapter {
138			inner: Arc::new(AdapterInner { ptr })
139		})
140	}
141
142	/// Loads an [`Adapter`] from memory. The adapter's values will be **copied**, either to the CPU or the given
143	/// allocator if one is provided.
144	///
145	/// An [`Allocator`] can be provided to specify the device on which the inputs should be allocated on.
146	/// Note that providing a CPU allocator will return an error; only device allocators are expected.
147	///
148	/// ```
149	/// # use ort::{
150	/// # 	ep,
151	/// # 	memory::DeviceType,
152	/// # 	session::{Adapter, RunOptions, Session},
153	/// # 	value::Tensor
154	/// # };
155	/// # fn main() -> ort::Result<()> {
156	/// let mut model = Session::builder()?
157	/// 	.with_execution_providers([ep::CUDA::default().build()])?
158	/// 	.commit_from_file("tests/data/lora_model.onnx")?;
159	///
160	/// let bytes = std::fs::read("tests/data/adapter.orl").unwrap();
161	/// let allocator = model.allocator();
162	/// let lora = Adapter::from_memory(
163	/// 	&bytes,
164	/// 	if allocator.memory_info().device_type() == DeviceType::CPU { None } else { Some(&allocator) }
165	/// )?;
166	///
167	/// let mut run_options = RunOptions::new()?;
168	/// run_options.add_adapter(&lora)?;
169	///
170	/// let outputs =
171	/// 	model.run_with_options(ort::inputs![Tensor::<f32>::from_array(([4, 4], vec![1.0; 16]))?], &run_options)?;
172	/// # Ok(())
173	/// # }
174	/// ```
175	pub fn from_memory(bytes: &[u8], allocator: Option<&Allocator>) -> Result<Self> {
176		let allocator_ptr = allocator.map(|c| c.ptr().cast_mut()).unwrap_or_else(ptr::null_mut);
177		let mut ptr = ptr::null_mut();
178		ortsys![unsafe CreateLoraAdapterFromArray(bytes.as_ptr().cast(), bytes.len(), allocator_ptr, &mut ptr)?; nonNull(ptr)];
179		crate::logging::create!(Adapter, ptr);
180		Ok(Adapter {
181			inner: Arc::new(AdapterInner { ptr })
182		})
183	}
184}

202	#[test]
203	#[cfg(feature = "std")]
204	fn test_lora() -> crate::Result<()> {
205		let model = std::fs::read("tests/data/lora_model.onnx").expect("");
206		let mut session = Session::builder()?.commit_from_memory(&model)?;
207		let lora = Adapter::from_file("tests/data/adapter.orl", None)?;
208
209		let mut run_options = RunOptions::new()?;
210		run_options.add_adapter(&lora)?;
211
212		let output: Tensor<f32> = session
213			.run_with_options(crate::inputs![Tensor::<f32>::from_array(([4, 4], vec![1.0; 16]))?], &run_options)?
214			.remove("output")
215			.expect("")
216			.downcast()?;
217		let (_, output) = output.extract_tensor();
218		assert_eq!(output[0], 154.0);
219		assert_eq!(output[1], 176.0);
220		assert_eq!(output[2], 198.0);
221		assert_eq!(output[3], 220.0);
222
223		Ok(())
224	}
