70/// Holds onto an [`ort_sys::OrtSession`] pointer and its associated allocator.
71///
72/// Internally, this is wrapped in an [`Arc`] and shared between a [`Session`] and any [`Value`]s created as a result
73/// of [`Session::run`] to ensure that the [`Value`]s are kept alive until all references to the session are dropped.
74#[derive(Debug)]
75pub struct SharedSessionInner {
76	session_ptr: NonNull<ort_sys::OrtSession>,
77	pub(crate) allocator: Allocator,
78	/// Additional things we may need to hold onto for the duration of this session, like `OperatorDomain`s and
79	/// DLL handles for operator libraries.
80	_extras: SmallVec<[Arc<dyn Any>; 4]>,
81	_environment: Arc<Environment>
82}
83
84unsafe impl Send for SharedSessionInner {}
85unsafe impl Sync for SharedSessionInner {}
86
87impl AsPointer for SharedSessionInner {
88	type Sys = ort_sys::OrtSession;
89
90	fn ptr(&self) -> *const Self::Sys {
91		self.session_ptr.as_ptr()
92	}
93}
94
95impl Drop for SharedSessionInner {
96	fn drop(&mut self) {
97		ortsys![unsafe ReleaseSession(self.session_ptr.as_ptr())];
98		crate::logging::drop!(Session, self.session_ptr);
99	}
100}

173	/// Returns a list of initializers which are overridable (i.e. also graph inputs).
174	#[must_use]
175	pub fn overridable_initializers(&self) -> Vec<OverridableInitializer> {
176		// can only fail if:
177		// - index is out of bounds (impossible because of the loop)
178		// - the model is not loaded (how could this even be possible?)
179		let mut size = 0;
180		ortsys![unsafe SessionGetOverridableInitializerCount(self.ptr(), &mut size).expect("infallible")];
181		let allocator = Allocator::default();
182		(0..size)
183			.map(|i| {
184				let mut name: *mut c_char = ptr::null_mut();
185				ortsys![unsafe SessionGetOverridableInitializerName(self.ptr(), i, allocator.ptr().cast_mut(), &mut name).expect("infallible")];
186				let name = unsafe { CStr::from_ptr(name) }.to_string_lossy().into_owned();
187				let mut typeinfo_ptr: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
188				ortsys![unsafe SessionGetOverridableInitializerTypeInfo(self.ptr(), i, &mut typeinfo_ptr).expect("infallible"); nonNull(typeinfo_ptr)];
189				let dtype = unsafe { ValueType::from_type_info(typeinfo_ptr) };
190				OverridableInitializer { name, dtype }
191			})
192			.collect()
193	}

204	/// # fn main() -> ort::Result<()> {
205	/// let mut session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
206	/// let input = ndarray::Array4::<f32>::zeros((1, 64, 64, 3));

208	/// # 	Ok(())

212	pub fn run<'s, 'i, 'v: 'i, const N: usize>(&'s mut self, input_values: impl Into<SessionInputs<'i, 'v, N>>) -> Result<SessionOutputs<'s>> {
213		match input_values.into() {
214			SessionInputs::ValueSlice(input_values) => {
215				self.run_inner(self.inputs.iter().map(|input| input.name()).collect(), input_values.iter().collect(), None)
216			}
217			SessionInputs::ValueArray(input_values) => {
218				self.run_inner(self.inputs.iter().map(|input| input.name()).collect(), input_values.iter().collect(), None)
219			}
220			SessionInputs::ValueMap(input_values) => {
221				self.run_inner(input_values.iter().map(|(k, _)| k.as_ref()).collect(), input_values.iter().map(|(_, v)| v).collect(), None)
222			}
223		}
224	}

226	/// Run input data through the ONNX graph, performing inference, with a [`RunOptions`] struct. The most common usage
227	/// of `RunOptions` is to allow the session run to be terminated from a different thread.
228	///
229	/// ```no_run
230	/// # // no_run because upsample.onnx is too simple of a model for the termination signal to be reliable enough
231	/// # use std::sync::Arc;
232	/// # use ort::{session::{Session, RunOptions}, value::{Value, ValueType, TensorRef, TensorElementType}};
233	/// # fn main() -> ort::Result<()> {
234	/// # 	let mut session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
235	/// # 	let input = Value::from_array(ndarray::Array4::<f32>::zeros((1, 64, 64, 3)))?;
236	/// let run_options = Arc::new(RunOptions::new()?);
237	///
238	/// let run_options_ = Arc::clone(&run_options);
239	/// std::thread::spawn(move || {
240	/// 	let _ = run_options_.terminate();
241	/// });
242	///
243	/// let res = session.run_with_options(ort::inputs![&input], &*run_options);
244	/// // upon termination, the session will return an `Error::SessionRun` error.`
245	/// assert_eq!(
246	/// 	&res.unwrap_err().to_string(),
247	/// 	"Failed to run inference on model: Exiting due to terminate flag being set to true."
248	/// );
249	/// # 	Ok(())
250	/// # }
251	/// ```
252	#[cfg(not(target_arch = "wasm32"))]
253	pub fn run_with_options<'r, 's: 'r, 'i, 'v: 'i, O: SelectedOutputMarker, const N: usize>(
254		&'s mut self,
255		input_values: impl Into<SessionInputs<'i, 'v, N>>,
256		run_options: &'r RunOptions<O>
257	) -> Result<SessionOutputs<'r>> {
258		match input_values.into() {
259			SessionInputs::ValueSlice(input_values) => {
260				self.run_inner(self.inputs.iter().map(|input| input.name()).collect(), input_values.iter().collect(), Some(&run_options.inner))
261			}
262			SessionInputs::ValueArray(input_values) => {
263				self.run_inner(self.inputs.iter().map(|input| input.name()).collect(), input_values.iter().collect(), Some(&run_options.inner))
264			}
265			SessionInputs::ValueMap(input_values) => {
266				self.run_inner(input_values.iter().map(|(k, _)| k.as_ref()).collect(), input_values.iter().map(|(_, v)| v).collect(), Some(&run_options.inner))
267			}
268		}
269	}

271	#[cfg(not(target_arch = "wasm32"))]
272	fn run_inner<'i, 'r, 's: 'r, 'v: 'i>(
273		&'s self,
274		input_names: SmallVec<[&str; STACK_SESSION_INPUTS]>,
275		input_values: SmallVec<[&'i SessionInputValue<'v>; STACK_SESSION_INPUTS]>,
276		run_options: Option<&'r UntypedRunOptions>
277	) -> Result<SessionOutputs<'r>> {
278		if input_values.len() > input_names.len() {
279			// If we provide more inputs than the model expects with `ort::inputs![a, b, c]`, then we get an `input_names` shorter
280			// than `inputs`. ONNX Runtime will attempt to look up the name of all inputs before doing any checks, thus going out of
281			// bounds of `input_names` and triggering a segfault, so we check that condition here. This will never trip for
282			// `ValueMap` inputs since the number of names & values are always equal as its a vec of tuples.
283			return Err(Error::new_with_code(
284				ErrorCode::InvalidArgument,
285				format!("{} inputs were provided, but the model only accepts {}.", input_values.len(), input_names.len())
286			));
287		}
288
289		let (output_names, mut output_tensors) = match run_options {
290			Some(r) => r.outputs.resolve_outputs(&self.outputs),
291			None => (self.outputs.iter().map(|o| o.name()).collect(), iter::repeat_with(|| None).take(self.outputs.len()).collect())
292		};
293		let output_value_ptrs: SmallVec<[*mut ort_sys::OrtValue; STACK_SESSION_OUTPUTS]> = output_tensors
294			.iter_mut()
295			.map(|c| match c {
296				Some(v) => v.ptr_mut(),
297				None => ptr::null_mut()
298			})
299			.collect();
300		let input_value_ptrs: SmallVec<[*const ort_sys::OrtValue; STACK_SESSION_INPUTS]> = input_values.iter().map(|c| c.ptr()).collect();
301
302		let run_options_ptr = if let Some(run_options) = &run_options { run_options.ptr.as_ptr() } else { ptr::null() };
303
304		with_cstr_ptr_array(&input_names, &|input_name_ptrs| {
305			with_cstr_ptr_array(&output_names, &|output_name_ptrs| {
306				ortsys![
307					unsafe Run(
308						self.inner.session_ptr.as_ptr(),
309						run_options_ptr,
310						input_name_ptrs.as_ptr(),
311						input_value_ptrs.as_ptr(),
312						input_value_ptrs.len(),
313						output_name_ptrs.as_ptr(),
314						output_name_ptrs.len(),
315						output_value_ptrs.as_ptr().cast_mut()
316					)?
317				];
318				Ok(())
319			})
320		})?;
321
322		let outputs = output_tensors
323			.into_iter()
324			.enumerate()
325			.map(|(i, v)| match v {
326				Some(value) => value,
327				None => unsafe {
328					Value::from_ptr(
329						NonNull::new(output_value_ptrs[i]).expect("OrtValue ptr returned from session Run should not be null"),
330						Some(Arc::clone(&self.inner))
331					)
332				}
333			})
334			.collect();
335
336		Ok(SessionOutputs::new(output_names, outputs))
337	}

353	#[cfg(not(target_arch = "wasm32"))]
354	fn run_binding_inner<'r, 'b, 's: 'b>(
355		&'s self,
356		binding: &'b IoBinding,
357		run_options: Option<&'r RunOptions<NoSelectedOutputs>>
358	) -> Result<SessionOutputs<'b>> {
359		use crate::util::run_on_drop;
360
361		let run_options_ptr = if let Some(run_options) = run_options { run_options.ptr() } else { ptr::null() };
362		ortsys![unsafe RunWithBinding(self.inner.ptr().cast_mut(), run_options_ptr, binding.ptr())?];
363
364		let mut count = binding.output_values.len();
365		if count > 0 {
366			let mut output_values_ptr: *mut *mut ort_sys::OrtValue = ptr::null_mut();
367			ortsys![unsafe GetBoundOutputValues(binding.ptr(), self.inner.allocator.ptr().cast_mut(), &mut output_values_ptr, &mut count)?; nonNull(output_values_ptr)];
368
369			let _guard = run_on_drop(|| unsafe {
370				self.inner.allocator.free(output_values_ptr.as_ptr());
371			});
372			let output_values = unsafe { slice::from_raw_parts(output_values_ptr.as_ptr(), count) }
373				.iter()
374				.map(|ptr| unsafe {
375					DynValue::from_ptr(NonNull::new(*ptr).expect("OrtValue ptrs returned by GetBoundOutputValues should not be null"), Some(self.inner()))
376				})
377				.collect();
378
379			Ok(SessionOutputs::new(binding.output_values.iter().map(|(k, _)| k.as_str()).collect(), output_values))
380		} else {
381			Ok(SessionOutputs::new_empty())
382		}
383	}

385	/// Asynchronously run input data through the ONNX graph, performing inference.
386	///
387	/// Inference will be performed on a thread in the session's thread pool. **Thus, the session must have been
388	/// configured to have multiple intra-op threads**; see [`SessionBuilder::with_intra_threads`].
389	///
390	/// See [`crate::inputs!`] for a convenient macro which will help you create your session inputs from `ndarray`s or
391	/// other data. You can also provide a `Vec`, array, or `HashMap` of [`Value`]s if you create your inputs
392	/// dynamically.
393	///
394	/// ```
395	/// # use std::sync::Arc;
396	/// # use ort::{session::{Session, RunOptions}, value::{Value, ValueType, TensorRef, TensorElementType}};
397	/// # fn main() -> ort::Result<()> { tokio::runtime::Builder::new_current_thread().enable_all().build().unwrap().block_on(async {
398	/// let mut session = Session::builder()?.with_intra_threads(2)?.commit_from_file("tests/data/upsample.onnx")?;
399	/// let input = ndarray::Array4::<f32>::zeros((1, 64, 64, 3));
400	/// let options = RunOptions::new()?;
401	/// let outputs = session.run_async(ort::inputs![TensorRef::from_array_view(&input)?], &options)?.await?;
402	/// # 	Ok(())
403	/// # }) }
404	/// ```
405	#[cfg(all(feature = "std", not(target_arch = "wasm32")))]
406	#[cfg_attr(docsrs, doc(cfg(feature = "std")))]
407	pub fn run_async<'r, 's: 'r, 'i, 'v: 'i + 's, O: SelectedOutputMarker, const N: usize>(
408		&'s mut self,
409		input_values: impl Into<SessionInputs<'i, 'v, N>>,
410		run_options: &'r RunOptions<O>
411	) -> Result<InferenceFut<'r, 'v>> {
412		match input_values.into() {
413			SessionInputs::ValueSlice(input_values) => {
414				self.run_inner_async(self.inputs.iter().map(|input| input.name()).collect(), input_values.iter().collect(), &run_options.inner)
415			}
416			SessionInputs::ValueArray(input_values) => {
417				self.run_inner_async(self.inputs.iter().map(|input| input.name()).collect(), input_values.iter().collect(), &run_options.inner)
418			}
419			SessionInputs::ValueMap(input_values) => {
420				self.run_inner_async(input_values.iter().map(|(k, _)| k.as_ref()).collect(), input_values.iter().map(|(_, v)| v).collect(), &run_options.inner)
421			}
422		}
423	}

425	#[cfg(all(feature = "std", not(target_arch = "wasm32")))]
426	fn run_inner_async<'i, 'r, 's: 'r, 'v: 'i + 's>(
427		&'s self,
428		input_names: SmallVec<[&str; STACK_SESSION_INPUTS]>,
429		input_values: SmallVec<[&SessionInputValue<'v>; STACK_SESSION_INPUTS]>,
430		run_options: &'r Arc<UntypedRunOptions>
431	) -> Result<InferenceFut<'r, 'v>> {
432		let input_name_ptrs = input_names
433			.into_iter()
434			.map(|name| CString::new(name.as_bytes()).map(|s| s.into_raw().cast_const()))
435			.collect::<Result<SmallVec<[*const c_char; STACK_SESSION_INPUTS]>, _>>()?;
436
437		let mut input_inner_holders = SmallVec::with_capacity(input_values.len());
438		let mut input_ort_values = SmallVec::with_capacity(input_values.len());
439		for input in input_values {
440			input_ort_values.push(input.ptr());
441			input_inner_holders.push(Arc::clone(match input {
442				SessionInputValue::ViewMut(v) => &(**v).inner,
443				SessionInputValue::View(v) => &(**v).inner,
444				SessionInputValue::Owned(v) => &v.inner
445			}));
446		}
447
448		let (output_names, mut output_tensors) = run_options.outputs.resolve_outputs(&self.outputs);
449		let output_name_ptrs = output_names
450			.iter()
451			.map(|n| CString::new(*n).unwrap_or_else(|_| unreachable!()))
452			.map(|n| n.into_raw().cast_const())
453			.collect();
454		let output_tensor_ptrs = output_tensors
455			.iter_mut()
456			.map(|c| match c {
457				Some(v) => v.ptr_mut(),
458				None => ptr::null_mut()
459			})
460			.collect();
461
462		let async_inner = Arc::new(InferenceFutInner::new(Arc::clone(run_options)));
463
464		// Avoid creating AsyncInferenceContext on the stack since it is a very large struct. Instead, create it on the heap and
465		// then fill it with values.
466		let mut ctx = Box::<AsyncInferenceContext>::new_uninit();
467		unsafe {
468			use core::ptr::write;
469
470			let ctx = ctx.assume_init_mut();
471			write(&mut ctx.inner, Arc::clone(&async_inner));
472			// everything allocated within `run_inner_async` needs to be kept alive until we are certain inference has completed and
473			// ONNX Runtime no longer needs the data - i.e. when `async_callback` is called. `async_callback` will free all of
474			// this data just like we do in `run_inner`
475			write(&mut ctx.input_ort_values, input_ort_values);
476			write(&mut ctx._input_inner_holders, input_inner_holders);
477			write(&mut ctx.input_name_ptrs, input_name_ptrs);
478			write(&mut ctx.output_name_ptrs, output_name_ptrs);
479			write(&mut ctx.output_names, output_names);
480			write(&mut ctx.output_value_ptrs, output_tensor_ptrs);
481			write(&mut ctx.session_inner, &self.inner);
482		};
483		let ctx = Box::leak(unsafe { ctx.assume_init() });
484		crate::logging::create!(AsyncInferenceContext, ctx);
485
486		ortsys![
487			unsafe RunAsync(
488				self.inner.session_ptr.as_ptr(),
489				run_options.ptr.as_ptr(),
490				ctx.input_name_ptrs.as_ptr(),
491				ctx.input_ort_values.as_ptr(),
492				ctx.input_ort_values.len(),
493				ctx.output_name_ptrs.as_ptr(),
494				ctx.output_name_ptrs.len(),
495				ctx.output_value_ptrs.as_mut_ptr(),
496				Some(self::r#async::async_callback),
497				ctx as *mut _ as *mut ort_sys::c_void
498			)?
499		];
500
501		Ok(InferenceFut::new(async_inner))
502	}

514	/// let mut session = Session::builder()?.with_intra_threads(2)?.commit_from_file("tests/data/upsample.onnx")?;
515	/// let input = ndarray::Array4::<f32>::zeros((1, 64, 64, 3));
516	/// let options = RunOptions::new()?;
517	/// let outputs = session.run_async(ort::inputs![TensorRef::from_array_view(&input)?], &options)?.await?;
518	/// # 	Ok(())
519	/// # }) }
520	/// ```
521	#[cfg(target_arch = "wasm32")]
522	pub async fn run_async<'r, 's: 'r, 'i, 'v: 'i + 's, O: SelectedOutputMarker, const N: usize>(
523		&'s mut self,
524		input_values: impl Into<SessionInputs<'i, 'v, N>>,
525		run_options: &'r RunOptions<O>
526	) -> Result<SessionOutputs<'r>> {
527		match input_values.into() {
528			SessionInputs::ValueSlice(input_values) => {
529				self.run_inner_async(self.inputs.iter().map(|input| input.name()).collect(), input_values.iter().collect(), &run_options.inner)
530					.await
531			}
532			SessionInputs::ValueArray(input_values) => {
533				self.run_inner_async(self.inputs.iter().map(|input| input.name()).collect(), input_values.iter().collect(), &run_options.inner)
534					.await
535			}
536			SessionInputs::ValueMap(input_values) => {
537				self.run_inner_async(input_values.iter().map(|(k, _)| k.as_ref()).collect(), input_values.iter().map(|(_, v)| v).collect(), &run_options.inner)
538					.await
539			}
540		}
541	}
542
543	#[cfg(target_arch = "wasm32")]
544	async fn run_inner_async<'i, 'r, 's: 'r, 'v: 'i + 's>(
545		&'s self,
546		input_names: SmallVec<[&str; STACK_SESSION_INPUTS]>,
547		input_values: SmallVec<[&SessionInputValue<'v>; STACK_SESSION_INPUTS]>,
548		run_options: &'r UntypedRunOptions
549	) -> Result<SessionOutputs<'r>> {
550		if input_values.len() > input_names.len() {
551			return Err(Error::new_with_code(
552				ErrorCode::InvalidArgument,
553				format!("{} inputs were provided, but the model only accepts {}.", input_values.len(), input_names.len())
554			));
555		}
556
557		let (output_names, mut output_tensors) = run_options.outputs.resolve_outputs(&self.outputs);
558		let mut output_value_ptrs: SmallVec<[*mut ort_sys::OrtValue; STACK_SESSION_OUTPUTS]> = output_tensors
559			.iter_mut()
560			.map(|c| match c {
561				Some(v) => v.ptr_mut(),
562				None => ptr::null_mut()
563			})
564			.collect();
565		let input_value_ptrs: SmallVec<[*const ort_sys::OrtValue; STACK_SESSION_INPUTS]> = input_values.iter().map(|c| c.ptr()).collect();
566
567		let status = ortsys![
568			unsafe RunAsync(
569				self.inner.session_ptr.as_ptr(),
570				run_options.ptr.as_ptr(),
571				&input_names,
572				&input_value_ptrs,
573				&output_names,
574				&mut output_value_ptrs
575			)
576		]
577		.await;
578		unsafe { Error::result_from_status(status) }?;
579
580		let outputs = output_tensors
581			.into_iter()
582			.enumerate()
583			.map(|(i, v)| match v {
584				Some(value) => value,
585				None => unsafe {
586					Value::from_ptr(
587						NonNull::new(output_value_ptrs[i]).expect("OrtValue ptr returned from session Run should not be null"),
588						Some(Arc::clone(&self.inner))
589					)
590				}
591			})
592			.collect();
593
594		Ok(SessionOutputs::new(output_names, outputs))
595	}

620	/// Sets this session's [workload type][`WorkloadType`] to instruct execution providers to prioritize performance or
621	/// efficiency.
622	///
623	/// ```
624	/// # use std::sync::Arc;
625	/// # use ort::{session::{RunOptions, Session, WorkloadType}, value::{Value, ValueType, TensorRef, TensorElementType}};
626	/// # fn main() -> ort::Result<()> {
627	/// let mut session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
628	/// session.set_workload_type(WorkloadType::Efficient)?;
629	///
630	/// let input = ndarray::Array4::<f32>::zeros((1, 64, 64, 3));
631	/// let outputs = session.run(ort::inputs![TensorRef::from_array_view(&input)?])?;
632	/// # 	Ok(())
633	/// # }
634	/// ```
635	#[cfg(feature = "api-20")]
636	#[cfg_attr(docsrs, doc(cfg(feature = "api-20")))]
637	pub fn set_workload_type(&mut self, workload_type: WorkloadType) -> Result<()> {
638		static KEY: &[u8] = b"ep.dynamic.workload_type\0";
639		match workload_type {
640			WorkloadType::Default => self.set_dynamic_option(KEY.as_ptr().cast(), c"Default".as_ptr().cast()),
641			WorkloadType::Efficient => self.set_dynamic_option(KEY.as_ptr().cast(), c"Efficient".as_ptr().cast())
642		}
643	}

645	#[cfg(feature = "api-20")]
646	pub(crate) fn set_dynamic_option(&mut self, key: *const c_char, value: *const c_char) -> Result<()> {
647		ortsys![unsafe SetEpDynamicOptions(self.inner.session_ptr.as_ptr(), &key, &value, 1)?];
648		Ok(())
649	}

662/// Workload type, used to signal to execution providers whether to prioritize performance or efficiency.
663///
664/// See [`Session::set_workload_type`].
665#[derive(Default, Debug, Clone, Copy, PartialEq, Eq)]
666pub enum WorkloadType {
667	/// Prioritize performance. This is the default behavior when the workload type is not overridden.
668	#[default]
669	Default,
670	/// Prioritize efficiency, by i.e. reducing scheduling priority and/or offloading to efficiency cores.
671	Efficient
672}

715	fn extract_io_name(
716		f: unsafe extern "system" fn(*const ort_sys::OrtSession, usize, *mut ort_sys::OrtAllocator, *mut *mut c_char) -> ort_sys::OrtStatusPtr,
717		session_ptr: NonNull<ort_sys::OrtSession>,
718		allocator: &Allocator,
719		i: usize
720	) -> Result<String> {
721		let mut name_ptr: *mut c_char = ptr::null_mut();
722
723		let status = unsafe { f(session_ptr.as_ptr(), i, allocator.ptr().cast_mut(), &mut name_ptr) };
724		unsafe { Error::result_from_status(status) }?;

726			crate::util::cold();
727			return Err(crate::Error::new("expected `name_ptr` to not be null"));
728		}
729
730		unsafe { AllocatedString::from_ptr(name_ptr, allocator) }.map(|x| x.to_string())
731	}

735		session_ptr: NonNull<ort_sys::OrtSession>,
